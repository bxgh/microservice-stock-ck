class FractalMarketAnalyzer:
    def __init__(self):
        self.hurst_exponents = {}
        self.multifractal_spectrum = {}

    def calculate_hurst_exponent(self, tick_data, feature='price'):
        """
        计算赫斯特指数，检测市场的分形特征
        H > 0.5: 趋势性（持久性）
        H < 0.5: 均值回归（反持久性）
        H = 0.5: 随机游走
        """
        if feature == 'price':
            series = tick_data['price'].values
        elif feature == 'volume':
            series = tick_data['vol'].values

        # 重标极差法计算赫斯特指数
        n = len(series)
        max_lag = min(n // 4, 100)  # 最大滞后

        R_S_ratios = []
        lags = []

        for lag in range(10, max_lag + 1, 10):
            # 计算每个子序列的R/S
            rs_values = []
            for i in range(0, n - lag, lag):
                sub_series = series[i:i+lag]
                mean_val = np.mean(sub_series)
                deviations = sub_series - mean_val
                cumulative_deviations = np.cumsum(deviations)

                R = np.max(cumulative_deviations) - np.min(cumulative_deviations)  # 极差
                S = np.std(sub_series)  # 标准差

                if S > 0:
                    rs_values.append(R / S)

            if rs_values:
                R_S_ratios.append(np.mean(rs_values))
                lags.append(lag)

        # 线性回归求赫斯特指数
        if len(R_S_ratios) > 1:
            log_rs = np.log(R_S_ratios)
            log_lags = np.log(lags)
            hurst_exponent, _ = np.polyfit(log_lags, log_rs, 1)
        else:
            hurst_exponent = 0.5

        return hurst_exponent

    def multifractal_analysis(self, tick_data):
        """多重分形分析"""
        price_series = tick_data['price'].values

        # 使用MF-DFA方法计算多重分形谱

class DeepTickFeatureLearner:
    def __init__(self, input_dim, hidden_dims=[128, 64, 32]):
        self.autoencoder = self.build_autoencoder(input_dim, hidden_dims)
        self.feature_importance = {}

    def build_autoencoder(self, input_dim, hidden_dims):
        """构建自编码器用于无监督特征学习"""
        # 编码器
        encoder_layers = []
        current_dim = input_dim
        for hidden_dim in hidden_dims:
            encoder_layers.extend([
                nn.Linear(current_dim, hidden_dim),
                nn.ReLU(),
                nn.Dropout(0.2)
            ])
            current_dim = hidden_dim

        # 解码器（对称结构）
        decoder_layers = []
        for hidden_dim in reversed(hidden_dims[:-1]):
            decoder_layers.extend([
                nn.Linear(current_dim, hidden_dim),
                nn.ReLU(),
                nn.Dropout(0.2)
            ])
            current_dim = hidden_dim
        decoder_layers.append(nn.Linear(current_dim, input_dim))

        return nn.Sequential(*encoder_layers, *decoder_layers)

    def learn_latent_features(self, tick_features):
        """学习潜在特征表示"""
        # 训练自编码器
        trained_encoder = self.train_autoencoder(tick_features)

        # 提取潜在特征
        with torch.no_grad():
            latent_features = trained_encoder[:len(self.autoencoder)//2](tick_features)
