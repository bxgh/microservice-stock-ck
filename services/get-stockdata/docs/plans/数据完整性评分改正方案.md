# 📊 数据完整性评分改正方案

## 📋 改正方案概述

### **问题背景**
当前系统数据完整性评分24/100（E级）主要由以下原因导致：
1. **评估标准过于严格**：用24小时标准评估股票分笔数据
2. **时间覆盖度计算不合理**：分笔数据天然具有时段性
3. **数据密度评估不适用**：用连续时间序列标准评估事件驱动数据
4. **字段完整性有待提升**：mootdx接口字段有限

### **改正目标**
- **短期目标**：评分从24/100提升至70/100（C级）
- **中期目标**：评分提升至85/100（B级）
- **长期目标**：达到90/100以上（A级）

#### **权重重新分配**
- 业务相关性：60%
- 技术完整性：30%
- 扩展性：10%

    """修正后的时间覆盖度计算"""
    if not data_list:
        return 0.0

    # 定义标准交易时段
    trading_sessions = {
        'morning_auction': ('09:25:00', '09:30:00'),   # 集合竞价
        'morning_trading': ('09:30:00', '11:30:00'),    # 上午连续交易
        'lunch_break': ('11:30:00', '13:00:00'),      # 午间休市
        'afternoon_trading': ('13:00:00', '15:00:00')  # 下午连续交易
    }

    total_trading_minutes = 240  # 4小时 = 240分钟
    actual_covered_minutes = _calculate_covered_trading_time(data_list, trading_sessions)

    return min(100, (actual_covered_minutes / total_trading_minutes) * 100)

def calculate_data_density_correctly(tick_data):
    """修正后的数据密度评估"""
    if not tick_data:
        return 0.0

    # 基于历史数据建立密度基准
    volume = sum(tick.volume for tick in tick_data)
    time_span = _get_time_span(tick_data)

    # 业务密度指标：每分钟平均交易笔数
    if time_span > 0:
        density_per_minute = len(tick_data) / (time_span / 60)

        # 基于成交量调整密度期望
        volume_adjusted_expectation = _calculate_volume_based_expectation(volume)

        density_score = min(100, (density_per_minute / volume_adjusted_expectation) * 100)
    else:
        density_score = 50.0  # 默认中等分数

    return density_score

```python
class MootdxFieldEnhancer:
    def __init__(self):
        self.tick_history = []
        self.price_history = []
        self.volume_history = []
        self.direction_history = []

    def enhance_tick_data(self, raw_data: List[List]) -> List[Dict]:
        """增强mootdx原始数据，计算衍生字段"""
        enhanced_data = []

        for i, tick in enumerate(raw_data):
            # 基础字段映射
            enhanced_tick = {
                'time': self._parse_time(tick[0]),
                'price': float(tick[1]),
                'volume': int(tick[2]),
                'amount': float(tick[3]),
                'direction': self._convert_direction(tick[4]),
                'code': '',  # 外部传入
                'date': '',  # 外部传入
            }

            # 衍生字段计算
            enhanced_tick.update(self._calculate_price_fields(float(tick[1]), i))
            enhanced_tick.update(self._calculate_volume_fields(int(tick[2]), i))
            enhanced_tick.update(self._calculate_direction_fields(tick[4], i))
            enhanced_tick.update(self._calculate_microstructure_fields(i))

            enhanced_data.append(enhanced_tick)

            # 更新历史数据
            self._update_history(enhanced_tick)

        return enhanced_data

    def _calculate_price_fields(self, current_price: float, index: int) -> Dict[str, Any]:
        """计算价格相关衍生字段"""
        if index > 0 and self.price_history:
            prev_price = self.price_history[-1]
            price_change = current_price - prev_price
            price_change_pct = (price_change / prev_price) * 100

            return {
                'price_change': round(price_change, 4),
                'price_change_pct': round(price_change_pct, 2),
                'price_high': max(current_price, max(self.price_history)),
                'price_low': min(current_price, min(self.price_history))
            }
        else:
            return {
                'price_change': 0.0,
                'price_change_pct': 0.0,
                'price_high': current_price,
                'price_low': current_price
            }

    def _calculate_volume_fields(self, current_volume: int, index: int) -> Dict[str, Any]:
        """计算成交量相关衍生字段"""
        cumulative_volume = sum(self.volume_history) + current_volume

        fields = {
            'cumulative_volume': cumulative_volume,
            'volume_ratio': round(current_volume / max(cumulative_volume, 1), 4)
        }

        # 移动平均计算
        if len(self.volume_history) >= 10:
            recent_volumes = self.volume_history[-9:] + [current_volume]
            fields['volume_ma_10'] = round(sum(recent_volumes) / 10, 0)
        else:
            fields['volume_ma_10'] = current_volume

        # 大单标识
        if len(self.volume_history) > 10:
            avg_volume = sum(self.volume_history[-20:]) / 20
            fields['is_large_order'] = current_volume > avg_volume * 2
        else:
            fields['is_large_order'] = False

        return fields
```

#### **一致性检查规则**
```python
class DataConsistencyValidator:
    def __init__(self):
        self.consistency_rules = [
            self._validate_price_amount_consistency,
            self._validate_time_date_consistency,
            self._validate_direction_volume_consistency,
            self._validate_price_range_reasonability
        ]

    def validate_tick_data(self, tick_data: List[Dict]) -> Dict[str, Any]:
        """验证分笔数据一致性"""
        if not tick_data:
            return {'score': 0, 'issues': ['无数据']}

        issues = []
        total_checks = 0
        passed_checks = 0

        for rule in self.consistency_rules:
            total_checks += 1
            rule_result = rule(tick_data)
            if rule_result['passed']:
                passed_checks += 1
            else:
                issues.extend(rule_result['issues'])

        consistency_score = (passed_checks / total_checks) * 100

        return {
            'score': round(consistency_score, 1),
            'issues': issues,
            'passed_checks': passed_checks,
            'total_checks': total_checks
        }

    def _validate_price_amount_consistency(self, tick_data: List[Dict]) -> Dict[str, Any]:
        """验证价格与成交额的一致性"""
        inconsistencies = []

        for i, tick in enumerate(tick_data):
            price = tick.get('price', 0)
            volume = tick.get('volume', 0)
            amount = tick.get('amount', 0)

            if volume > 0:
                expected_amount = price * volume
                actual_amount = amount

                # 允许5%的误差范围（考虑四舍五入）
                tolerance = expected_amount * 0.05
                if abs(actual_amount - expected_amount) > tolerance:
                    inconsistencies.append(
                        f"Tick {i}: 成交额不一致 (预期:{expected_amount:.2f}, 实际:{actual_amount:.2f})"
                    )

        return {
            'passed': len(inconsistencies) == 0,
            'issues': inconsistencies
        }
```

### **3.1 分层评分模型实现**

#### **评分权重分配**
```python
class ImprovedCompletenessScorer:
    def __init__(self):
        self.weights = {
            'basic_layer': 0.40,    # 基础层：字段完整性
            'business_layer': 0.40,  # 业务层：数据准确性
            'quality_layer': 0.20   # 质量层：数据质量
        }

    def calculate_comprehensive_score(self, tick_data: List[TickData]) -> Dict[str, Any]:
        """计算综合完整性评分"""
        # 基础层评分
        basic_score = self._evaluate_basic_completeness(tick_data)

        # 业务层评分
        business_score = self._evaluate_business_value(tick_data)

        # 质量层评分
        quality_score = self._evaluate_data_quality(tick_data)

        # 综合评分
        comprehensive_score = (
            basic_score * self.weights['basic_layer'] +
            business_score * self.weights['business_layer'] +
            quality_score * self.weights['quality_layer']
        )

        return {
            'overall_score': round(comprehensive_score, 1),
            'grade': self._determine_grade(comprehensive_score),
            'basic_score': round(basic_score, 1),
            'business_score': round(business_score, 1),
            'quality_score': round(quality_score, 1),
            'details': {
                'total_records': len(tick_data),
                'time_coverage': self._get_time_coverage(tick_data),
                'field_completeness': self._get_field_completeness(tick_data),
                'consistency_issues': self._get_consistency_issues(tick_data)
            }
        }

    def _evaluate_basic_completeness(self, tick_data: List[TickData]) -> float:
        """评估基础层完整性"""
        if not tick_data:
            return 0.0

        # 必填字段检查
        required_fields = ['time', 'price', 'volume', 'amount', 'direction']
        field_completeness = 0.0

        for tick in tick_data:
            tick_completeness = 0
            for field in required_fields:
                if hasattr(tick, field) and getattr(tick, field) is not None:
                    tick_completeness += 1

            field_completeness += tick_completeness / len(required_fields)

        return (field_completeness / len(tick_data)) * 100

    def _evaluate_business_value(self, tick_data: List[TickData]) -> float:
        """评估业务层价值"""
        if not tick_data:
            return 0.0

        business_metrics = {
            'time_span_score': self._evaluate_time_span(tick_data),
            'data_density_score': self._evaluate_data_density(tick_data),
            'price_quality_score': self._evaluate_price_quality(tick_data),
            'volume_quality_score': self._evaluate_volume_quality(tick_data)
        }

        # 业务指标权重
        business_weights = {
            'time_span_score': 0.30,
            'data_density_score': 0.25,
            'price_quality_score': 0.25,
            'volume_quality_score': 0.20
        }

        business_score = sum(
            metric * business_weights[metric_name]
            for metric_name, metric in business_metrics.items()
        )

        return min(100, business_score)

    def _evaluate_data_quality(self, tick_data: List[TickData]) -> float:
        """评估质量层质量"""
        if not tick_data:
            return 0.0

        quality_metrics = {
            'consistency_score': self._calculate_consistency_score(tick_data),
            'accuracy_score': self._calculate_accuracy_score(tick_data),
            'completeness_score': self._calculate_enhanced_completeness(tick_data)
        }

        # 质量指标权重
        quality_weights = {
            'consistency_score': 0.40,
            'accuracy_score': 0.35,
            'completeness_score': 0.25
        }

        quality_score = sum(
            metric * quality_weights[metric_name]
            for metric_name, metric in quality_metrics.items()
        )

        return min(100, quality_score)
```

### **3.2 动态权重调整机制**

#### **场景化权重配置**
```python
class DynamicWeightManager:
    def __init__(self):
        self.scenario_weights = {
            'real_time_trading': {
                'basic_layer': 0.25,
                'business_layer': 0.50,
                'quality_layer': 0.25
            },
            'historical_analysis': {
                'basic_layer': 0.50,
                'business_layer': 0.30,
                'quality_layer': 0.20
            },
            'risk_control': {
                'basic_layer': 0.30,
                'business_layer': 0.30,
                'quality_layer': 0.40
            }
        }

    def get_adjusted_weights(self, scenario: str) -> Dict[str, float]:
        """根据场景获取调整后的权重"""
        base_weights = {
            'basic_layer': 0.40,
            'business_layer': 0.40,
            'quality_layer': 0.20
        }

        scenario_weights = self.scenario_weights.get(scenario, base_weights)

        # 验证权重总和
        total_weight = sum(scenario_weights.values())
        if abs(total_weight - 1.0) > 0.001:
            # 标准化权重
            scenario_weights = {
                k: v / total_weight for k, v in scenario_weights.items()
            }

        return scenario_weights
```
