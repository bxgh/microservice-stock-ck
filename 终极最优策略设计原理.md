```python
def calculate_smart_delay(request_time, data_count):
    # 基于请求时间
    if request_time < 0.1:      # 响应很快
        base_delay = 0.05s
    elif request_time < 0.3:    # 响应正常
        base_delay = 0.1s
    else:                       # 响应较慢
        base_delay = 0.2s

    # 基于数据量（避免服务器压力）
    if data_count > 5:
        base_delay *= 1.5

    return min(base_delay, 0.5)  # 最大0.5秒
```

### 数据新颖性检查

```python
def check_data_novelty(new_data, existing_data):
    # 多维度比较：时间、价格、成交量
    new_times = set(new_data['time'])

    for data in existing_data:
        existing_times = set(data['time'])
        overlap = len(new_times & existing_times)

        # 重叠度 > 80% 视为重复数据
        if overlap > len(new_times) * 0.8:
            return False

    return True
```
